class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        count = 0
        vowel = set('aeiou')
        for v in range(k):
            if s[v] in vowel:
                count += 1
        maxCount = count
        for i in range(k, len(s)):
            if s[i] in vowel:
                count += 1
            
            if s[i-k] in vowel:
                count -= 1
            
            if count > maxCount:
                maxCount = count
        return maxCount
