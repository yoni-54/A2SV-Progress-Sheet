class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        result = set()
        left = 0
        maxLength = 0
        for right in range(len(s)):
            while s[right] in result:
                result.remove(s[left])
                left += 1

            result.add(s[right])
            maxLength = max(maxLength, right - left + 1)
        return maxLength
