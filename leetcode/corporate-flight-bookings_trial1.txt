class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        arr = [0] * n

        for first, last, seats in bookings:
            start = first - 1
            end = last - 1

            arr[start] += seats

            if end + 1 < n:
                arr[end + 1] -= seats

        for i in range(1, n):
            arr[i] += arr[i - 1]

        return arr
