class Solution:
    def countGood(self, nums, k):
        freq = defaultdict(int)
        left = 0
        pairs = 0
        result = 0

        for right in range(len(nums)):
            pairs += freq[nums[right]]
            freq[nums[right]] += 1

            while pairs >= k:
                result += len(nums) - right  

                freq[nums[left]] -= 1
                pairs -= freq[nums[left]]
                left += 1

        return result
