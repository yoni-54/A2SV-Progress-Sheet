class Solution:
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        prefixCounts = defaultdict(int)
        prefixCounts[0] = 1  

        oddCount = 0
        result = 0
    
        for num in nums:
            if num % 2 == 1:
               oddCount += 1
        
            if (oddCount - k) in prefixCounts:
               result += prefixCounts[oddCount - k]
        
            prefixCounts[oddCount] += 1
    
        return result
