class Solution:
    def predictTheWinner(self, nums: List[int]) -> bool:
        memo = {}

        def dp(i, j):
            if i == j:
                return nums[i]

            if (i, j) in memo:
                return memo[(i, j)]

            take_left = nums[i] - dp(i + 1, j)
            take_right = nums[j] - dp(i, j - 1)

            memo[(i, j)] = max(take_left, take_right)
            return memo[(i, j)]

        return dp(0, len(nums) - 1) >= 0
