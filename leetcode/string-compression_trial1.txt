class Solution:
    def compress(self, chars: List[str]) -> int:
        i = 0
        write = 0

        while i < len(chars):
            j = i

            while j < len(chars) and chars[i] == chars[j]:
                j += 1

            chars[write] = chars[i]
            write += 1

            count = j - i

           
            if count > 1:
                for c in str(count):
                    chars[write] = c
                    write += 1

            i = j
            
        return write
