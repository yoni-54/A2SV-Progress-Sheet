class Solution:
    def maximumSubarraySum(self, nums: List[int], k: int) -> int:
        left = 0
        currentSum = 0
        maxSum = 0
        seen = set()

        for right in range(len(nums)):
            while nums[right] in seen:
                seen.remove(nums[left])
                currentSum -= nums[left]
                left += 1
            seen.add(nums[right])
            currentSum += nums[right]

            if right - left + 1 > k:
                seen.remove(nums[left])
                currentSum -= nums[left]
                left += 1

            if right - left + 1 == k:
                maxSum = max(maxSum, currentSum)

        return maxSum
