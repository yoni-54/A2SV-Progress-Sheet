class Solution:
    def timeRequiredToBuy(self, tickets: List[int], k: int) -> int:
        target = tickets[k]
        total = 0
    
        for i in range(len(tickets)):
            if i <= k:
               total += min(tickets[i], target)
            else:
               total += min(tickets[i], target - 1)
    
        return total
